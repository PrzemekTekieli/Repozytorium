#include<iostream>
#include<cstdlib>
using namespace std;/* namespace - 6 */
 /*pusta linijka - 7*/
void sortowanie_babelkowe(int tab[],int n)
{/* koemntarz 1*/
  for(int i=0;i<n;i++)/* for - 4*/
    for(int j=1;j<n-i;j++) //pêtla wewnêtrzna
    if(tab[j-1]>tab[j])/* randomowy if - 5*/
      //zamiana miejscami
      swap(tab[j-1], tab[j]);/* swap swap - 9*/
}
 
int main()/* oto funkcja main - 2*/
{/* poczatek klamrowy - 8*/
  int *tab, n;/* jakis int - 3*/
 
  cout<<"Ile liczb bêdziesz chcia³ posortowaæ? ";
  cin>>n;
 
  tab = new int [n]; //przydzielenie pamiêci na elementy tablicy
  //wczytanie liczb
  for(int i=0;i<n;i++)
    cin>>tab[i];
 
  sortowanie_babelkowe(tab,n);
 
  //wypisanie posortowanych elementów
  for(int i=0;i<n;i++)
          cout<<tab[i]<<" ";
 
  cout<<endl;
  system("pause");
  return 0;
}
