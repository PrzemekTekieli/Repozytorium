void sortowanie_babelkowe(int tab[],int n)
{/* koemntarz 1*/
  for(int i=0;i<n;i++)/* for - 4*/
    for(int j=1;j<n-i;j++) //pêtla wewnêtrzna
    if(tab[j-1]>tab[j])/* if - 10 */
      //zamiana miejscami
      swap(tab[j-1], tab[j]);/* swap swap - 9*/
}
